    private void DisplayData(int ID, String user_email, String user_name) throws FileNotFoundException {
        File file = new File("Buy.txt");
            Scanner inputFile;
            int price = 1;
            String user_phone="",chosen_Movie="",movie_Type="",translation="",pay_method="";
            inputFile = new Scanner(file);
            while(inputFile.hasNext()){
                int id=inputFile.nextInt();
                if(ID==id){
                    inputFile.next();
                    user_phone=inputFile.next();
                    movie_Type=inputFile.next();
                    chosen_Movie=inputFile.next();
                    translation=inputFile.next();
                    pay_method=inputFile.next();
                    break;  
                }
                inputFile.nextLine();
            }
            inputFile.close();
            
            file = new File("BuyPrice.txt");
            inputFile = new Scanner(file);
            while(inputFile.hasNext()){
                String movie_type=inputFile.next();
                String chosen_Movie_file=inputFile.next();
                if(movie_Type.equals(movie_type) &&chosen_Movie.equals(chosen_Movie_file)){
                   price=inputFile.nextInt();
                    break;
                }
              inputFile.nextLine();
            }
            inputFile.close();
            name.setText(user_name);
            email.setText(user_email);
            phone.setText(user_phone);
            movieType.setText(movie_Type);
            chosenMovie.setText(chosen_Movie);
            Translation.setText(translation);
            paymethod.setText(pay_method);
            Price.setText(String.valueOf(price));
    
    }
